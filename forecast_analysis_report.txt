    def calculate_realistic_forecast(monthly_revenues):
        # Remove partial months
        full_months = remove_partial_months(monthly_revenues)

        if len(full_months) < 6:
            return "Insufficient data for reliable forecast"

        # Calculate robust growth rate
        growth_rates = calculate_mom_growth(full_months)
        median_growth = np.median(growth_rates)
        capped_growth = np.clip(median_growth, -0.10, 0.10)  # ±10% max

        # Calculate weighted baseline
        last_3_avg = np.mean(full_months[-3:])
        last_6_avg = np.mean(full_months[-6:])
        all_time_avg = np.mean(full_months)

        baseline = (0.5 * last_3_avg + 0.3 * last_6_avg + 0.2 * all_time_avg)

        # Generate forecasts
        forecasts = []
        for months_ahead in [1, 2, 3]:
            conservative = baseline * (1 + 0.5 * capped_growth) ** months_ahead
            moderate = baseline * (1 + capped_growth) ** months_ahead
            optimistic = baseline * (1 + min(1.5 * capped_growth, 0.10)) ** months_ahead

            # Calculate confidence intervals based on historical volatility
            historical_std = np.std(full_months)
            confidence_range = historical_std * (1 + 0.2 * months_ahead)

            forecasts.append({
                'month': months_ahead,
                'conservative': conservative,
                'moderate': moderate,
                'optimistic': optimistic,
                'lower_bound': moderate - confidence_range,
                'upper_bound': moderate + confidence_range
            })

        return forecasts
